DROP TABLE review_series, ongoing_series, finished_series, watchlist_series, series, review_movies, ongoing_movies, finished_movies, watchlist_movies, movies;

CREATE TABLE movies (
    movies_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50),
    duration INT,
    release_year INT,
    genre VARCHAR(30), 
    synopsis TEXT
);

CREATE TABLE watchlist_movies (
    movies_id INT,
    FOREIGN KEY (movies_id) REFERENCES movies(movies_id)
);

CREATE TABLE finished_movies (
    movies_id INT,
    finished_date DATE,
    FOREIGN KEY (movies_id) REFERENCES movies(movies_id)
);

CREATE TABLE ongoing_movies (
    movies_id INT,
    watched_duration INT,
    FOREIGN KEY (movies_id) REFERENCES movies(movies_id)
);

CREATE TABLE review_movies (
    movies_id INT,
    rating INT,
    FOREIGN KEY (movies_id) REFERENCES finished_movies(movies_id)
);

CREATE TABLE series (
    series_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(50),
    duration INT,
    release_year INT,
    genre VARCHAR(30), 
    synopsis TEXT,
    season INT,
    episode INT
);

CREATE TABLE watchlist_series (
    series_id INT,
    FOREIGN KEY (series_id) REFERENCES series(series_id)
);

CREATE TABLE finished_series (
    series_id INT,
    finished_date DATE,
    FOREIGN KEY (series_id) REFERENCES series(series_id)
);

CREATE TABLE ongoing_series (
    series_id INT,
    season_progress INT,
    episode_progress INT,
    watched_duration INT,
    FOREIGN KEY (series_id) REFERENCES series(series_id)
);

CREATE TABLE review_series (
    series_id INT,
    rating INT,
    review TEXT,
    FOREIGN KEY (series_id) REFERENCES finished_series(series_id)
);
